\section*{Problem 4}
$T(n) = 4T\frac{n}{2}+n, T(1) = 1$\\
$a = 4, b = 2, d = 0, f(n) = n \therefore{} n^{log_{b}^{d}} = n^{log_{2}^{4}}$\\
$=n^{2log^{2}_{2}}$
$=n^{2*1}$\\
=$n^2$\\
$\therefore{} T(n) = \theta(n^2)$\\

$T(n) = 4T\frac{n}{2}+n^2, T(1) = 1$\\
$a = 4, b = 2, d = 2, f(n)=n^2$\\
$n^{d}log(n) = n^{2}log(n)$\\
$\therefore{} T(n) = \theta(n^{2}log(n))$\\
 
$T(n) = 4T\frac{n}{2}+n^3, T(1) = 1$\\
$a = 4, b = 2, d = 3, f(n) = n^{3}$\\
$n^{d} = n^{3}$\\
$\therefore{} T(n) = \theta(n^{3})$
